暴力计算圆周率（蒙德卡洛方法）

标签（空格分隔）： 自编程序

from random import random
from time import perf_counter
DARTS=10001000100
hits=0.0
start=perf_counter()
for i in range(1,DARTS+1):
x,y=random(),random()
dist=pow(x2+y2,0.5)
if dist<=1.0:
hits=hits+1
pi=4*(hits/DARTS)
print(“圆周率值是:{}”.format(pi))
print(“运行时间是:{:.5f}s”.format(perf_counter()-start))
